    **Step 1: Add two routes to `autopilot/routes/api.js`**

    At the end of the file (after the cost/stats route, before any trailing `});`), add:

    ```javascript
    /**
     * GET /api/audit — full audit state (page-audit.json)
     * Returns all slugs with scores, issues, signals.
     */
    apiRouter.get('/audit', (req, res) => {
      const auditPath = join(STATE_DIR, 'page-audit.json');
      if (!existsSync(auditPath)) return res.json({ pages: {} });
      try {
        const data = JSON.parse(readFileSync(auditPath, 'utf8'));
        res.json(data);
      } catch (err) {
        logger.error({ err: err.message }, 'Failed to read page-audit.json');
        res.json({ pages: {} });
      }
    });

    /**
     * GET /api/audit/:slug — single page audit record
     * Returns 404 if slug not in audit data.
     */
    apiRouter.get('/audit/:slug', (req, res) => {
      const auditPath = join(STATE_DIR, 'page-audit.json');
      if (!existsSync(auditPath)) return res.status(404).json({ error: 'No audit data' });
      try {
        const data = JSON.parse(readFileSync(auditPath, 'utf8'));
        const record = data[req.params.slug];
        if (!record) return res.status(404).json({ error: 'Slug not found' });
        res.json(record);
      } catch (err) {
        logger.error({ err: err.message }, 'Failed to read audit for slug');
        res.status(500).json({ error: 'Failed to read audit data' });
      }
    });
    ```

    Note: `existsSync` and `readFileSync` are already imported at the top of api.js. `STATE_DIR` and `logger` are already defined. No new imports needed.

    **IMPORTANT:** Place the `/audit/:slug` route AFTER the `/audit` route to avoid Express treating "audit" as a slug param.
