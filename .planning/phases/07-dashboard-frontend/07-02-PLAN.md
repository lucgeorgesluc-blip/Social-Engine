```json

    {
      "keyword": "magnétiseuse Troyes",
      "data": [
        { "date": "2026-03-01", "position": 8 },
        { "date": "2026-03-15", "position": 5 }
      ],
      "currentPosition": 5,
      "delta": -3
    }
  ],
  "publishDates": [
    { "date": "2026-03-15", "slug": "auriculotherapie-arret-tabac", "title": "Auriculothérapie..." }
  ],
  "error": null
}
```
`error` is a non-null string when GSC credentials are missing — show empty state with the error message.
Period param: `7d` | `30d` | `90d` — defaults to `30d`.

### GET /api/links (auth required)
Response: D3-hierarchy-consumable JSON tree (built by `buildLinkTree()` from content-map.yaml):
```json
{
  "name": "Root",
  "type": "root",
  "children": [
    {
      "name": "Cluster: Magnétisme",
      "type": "pillar",
      "slug": "magnetiseur-troyes",
      "links_to": ["stop-tabac-troyes", "auriculotherapie-arret-tabac"],
      "children": [
        { "name": "Arrêt tabac Troyes", "type": "blog", "slug": "stop-tabac-troyes", "links_to": [], "children": [] },
        { "name": "Auriculothérapie...", "type": "blog", "slug": "auriculotherapie-arret-tabac", "links_to": [], "children": [] }
      ]
    }
  ]
}
```
Node types: `"pillar"` | `"service"` | `"blog"` | `"root"`
Orphan detection is CLIENT-SIDE — count inbound links per slug by traversing the tree.

```javascript
Alpine.data('rankingsPanel', () => ({
  loading: true,
  error: '',
  period: '30d',
  keywords: [],
  chart: null,

  COLORS: ['#3b82f6', '#22c55e', '#f59e0b', '#06b6d4', '#ec4899', '#a78bfa'],

  async init() {
    await this.loadData();
  },

  async loadData() {
    this.loading = true;
    this.error = '';
    try {
      const data = await apiFetch(`/api/rankings?period=${this.period}`);
      if (!data) return;
      if (data.error) {
        this.error = `Données GSC indisponibles : ${data.error}`;
        this.loading = false;
        return;
      }
      this.keywords = data.keywords || [];
      this.$nextTick(() => this.renderChart(data));
    } catch (e) {
      this.error = 'Erreur de chargement. Vérifiez votre connexion et rechargez la page.';
    } finally {
      this.loading = false;
    }
  },

  async setPeriod(p) {
    this.period = p;
    if (this.chart) { this.chart.destroy(); this.chart = null; }
    await this.loadData();
  },

  renderChart(data) {
    const canvas = document.getElementById('rankingsChart');
    if (!canvas) return;
    if (this.chart) this.chart.destroy();

    // Build annotation objects for each publish date
    const annotations = {};
    (data.publishDates || []).forEach((pub, i) => {
      annotations[`pub_${i}`] = {
        type: 'line',
        scaleID: 'x',
        value: pub.date,
        borderColor: 'rgba(59, 130, 246, 0.5)',
        borderWidth: 1,
        borderDash: [4, 4],
        label: {
          display: true,
          content: pub.title ? pub.title.substring(0, 20) + '…' : 'Publié',
          position: 'start',
          color: '#7d8590',
          font: { size: 11 }
        }
      };
    });

    this.chart = new Chart(canvas, {
      type: 'line',
      data: {
        datasets: (data.keywords || []).map((kw, i) => ({
          label: kw.keyword,
          data: (kw.data || []).map(d => ({ x: d.date, y: d.position })),
          borderColor: this.COLORS[i % this.COLORS.length],
          backgroundColor: 'transparent',
          tension: 0.3,
          pointRadius: 3,
          pointHoverRadius: 5,
        }))
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        color: '#7d8590',
        scales: {
          x: {
            type: 'time',
            time: { unit: 'day', displayFormats: { day: 'dd MMM' } },
            grid: { color: 'rgba(48, 54, 61, 0.8)' },
            ticks: { color: '#7d8590' }
          },
          y: {
            reverse: true,
            min: 1,
            grid: { color: 'rgba(48, 54, 61, 0.8)' },
            ticks: { color: '#7d8590' },
            title: { display: true, text: 'Position Google', color: '#7d8590' }
          }
        },
        plugins: {
          legend: {
            display: true,
            labels: { color: '#7d8590', boxWidth: 12 }
          },
          annotation: { annotations }
        }
      }
    });
  }
}));
```

**The complete `linksPanel` Alpine.data component:**

```javascript
Alpine.data('linksPanel', () => ({
  loading: true,
  error: '',
  treeData: null,
  selectedSlug: null,

  NODE_COLORS: {
    pillar:  '#f59e0b',
    service: '#3b82f6',
    blog:    '#06b6d4',
    root:    '#6b7280',
  },
  ORPHAN_COLOR: '#ef4444',

  async init() {
    try {
      const data = await apiFetch('/api/links');
      if (!data) return;
      this.treeData = data;
      this.$nextTick(() => this.renderTree(data));
    } catch {
      this.error = 'Erreur de chargement du maillage.';
    } finally {
      this.loading = false;
    }
  },

  getNodeColor(d, inboundCount) {
    const count = inboundCount[d.data.slug] || 0;
    if (d.data.type !== 'pillar' && d.data.type !== 'root' && count === 0) {
      return this.ORPHAN_COLOR;
    }
    return this.NODE_COLORS[d.data.type] || this.NODE_COLORS.blog;
  },

  renderTree(data) {
    const container = document.getElementById('treeContainer');
    if (!container) return;
    container.innerHTML = '';

    // Compute inbound link counts client-side
    const inboundCount = {};
    function countInbound(node) {
      (node.links_to || []).forEach(slug => {
        inboundCount[slug] = (inboundCount[slug] || 0) + 1;
      });
      (node.children || []).forEach(countInbound);
    }
    countInbound(data);

    const width = container.clientWidth || 900;
    const height = 500;
    const margin = { top: 20, right: 120, bottom: 20, left: 120 };

    const svg = d3.select(container)
      .append('svg')
      .attr('width', width)
      .attr('height', height)
      .style('background', 'transparent');

    const g = svg.append('g').attr('transform', `translate(${margin.left},${margin.top})`);

    // Zoom behavior
    svg.call(
      d3.zoom()
        .scaleExtent([0.3, 3])
        .on('zoom', (event) => g.attr('transform', event.transform))
    );

    const treeLayout = d3.tree().nodeSize([28, 200]);
    const self = this;
    let selectedSlug = null;

    function update(source) {
      const root = d3.hierarchy(data);

      // Restore collapsed state
      root.each(d => {
        if (d.data._collapsed) {
          d._children = d.children;
          d.children = null;
        }
      });

      treeLayout(root);

      // Links (edges)
      const links = g.selectAll('.link')
        .data(root.links(), d => d.target.data.slug || d.target.data.name);

      links.enter()
        .append('path')
        .attr('class', 'link')
        .merge(links)
        .attr('d', d3.linkHorizontal()
          .x(d => d.y)
          .y(d => d.x))
        .attr('fill', 'none')
        .attr('stroke', d => {
          if (!selectedSlug) return '#30363d';
          const src = d.source.data.slug;
          const tgt = d.target.data.slug;
          if (src === selectedSlug || tgt === selectedSlug) return '#3b82f6';
          return '#30363d';
        })
        .attr('stroke-width', d => {
          if (!selectedSlug) return 1.5;
          const src = d.source.data.slug;
          const tgt = d.target.data.slug;
          return (src === selectedSlug || tgt === selectedSlug) ? 2 : 1.5;
        })
        .attr('opacity', d => {
          if (!selectedSlug) return 1;
          const src = d.source.data.slug;
          const tgt = d.target.data.slug;
          return (src === selectedSlug || tgt === selectedSlug) ? 1 : 0.1;
        });

      links.exit().remove();

      // Nodes
      const nodes = g.selectAll('.node')
        .data(root.descendants(), d => d.data.slug || d.data.name);

      const nodeEnter = nodes.enter()
        .append('g')
        .attr('class', 'node')
        .attr('transform', d => `translate(${d.y},${d.x})`)
        .style('cursor', 'pointer')
        .on('click', (event, d) => {
          event.stopPropagation();
          if (d.data.slug === selectedSlug) {
            selectedSlug = null;
          } else {
            selectedSlug = d.data.slug;
            // Toggle collapse on nodes with children
            if (d.children || d._children) {
              if (d.children) {
                d.data._collapsed = true;
                d._children = d.children;
                d.children = null;
              } else {
                d.data._collapsed = false;
                d.children = d._children;
                d._children = null;
              }
            }
          }
          update(d);
        });

      nodeEnter.append('circle')
        .attr('r', d => d.data.type === 'pillar' ? 8 : 6)
        .attr('fill', d => self.getNodeColor(d, inboundCount))
        .attr('stroke', '#1c2128')
        .attr('stroke-width', 2);

      nodeEnter.append('text')
        .attr('dx', 10)
        .attr('dy', 4)
        .style('font-size', '12px')
        .style('font-family', 'system-ui, sans-serif')
        .style('fill', '#e6edf3')
        .text(d => {
          const name = d.data.name || d.data.slug || '';
          return name.length > 28 ? name.substring(0, 28) + '…' : name;
        });

      const nodeUpdate = nodeEnter.merge(nodes);
      nodeUpdate.attr('transform', d => `translate(${d.y},${d.x})`);
      nodeUpdate.select('circle')
        .attr('opacity', d => {
          if (!selectedSlug) return 1;
          if (d.data.slug === selectedSlug) return 1;
          // Check if connected
          const connected = root.links().some(l =>
            (l.source.data.slug === selectedSlug && l.target.data.slug === d.data.slug) ||
            (l.target.data.slug === selectedSlug && l.source.data.slug === d.data.slug)
          );
          return connected ? 1 : 0.3;
        });

      nodes.exit().remove();
    }

    update(null);

    // Click on SVG background resets selection
    svg.on('click', () => {
      selectedSlug = null;
      update(null);
    });
  }
}));
```

**Important implementation notes:**
- The D3 tree uses a horizontal layout (`d3.linkHorizontal()`) — nodes grow left-to-right
- `node._children` pattern stores collapsed children (D3 convention for collapsible trees)
- `data._collapsed` flag persists collapse state across `update()` calls since `d3.hierarchy()` is re-called each time
- `countInbound` traversal must handle both `children` (visible) and `_children` (collapsed) — use the raw `data` object fields (`node.children` from the JSON before D3 processes it)
- Orphan = `type !== 'pillar'` AND `type !== 'root'` AND `inboundCount[slug] === 0`
- SVG click handler on the background (not on nodes) resets `selectedSlug`
