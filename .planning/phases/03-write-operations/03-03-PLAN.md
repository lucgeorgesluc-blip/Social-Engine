```javascript
const yaml = require('js-yaml');
const fs = require('fs');
const path = require('path');

let _templates = null;

function getDmTemplates() {
  if (_templates) return _templates;
  try {
    const config = yaml.load(
      fs.readFileSync(path.resolve(__dirname, '../../.social-engine/config.yaml'), 'utf8')
    );
    var seq = config.dm_sequence || config.dm_templates || {};
    if (seq.templates || seq.messages) {
      _templates = seq.templates || seq.messages;
    }
  } catch (e) {
    console.error('[dm-templates] Could not load config.yaml:', e.message);
  }

  if (!_templates) {
    _templates = {
      msg1_sent: { label: 'Message 1 — Premier contact', text: 'Bonjour [Prenom], merci pour votre commentaire ! Souhaitez-vous en savoir plus ?' },
      msg2_sent: { label: 'Message 2 — Relance', text: 'Bonjour [Prenom], je me permets de revenir vers vous. Disponible pour un appel de 10 min ?' },
      msg3_sent: { label: 'Message 3 — Derniere relance', text: 'Bonjour [Prenom], si vous changez d\'avis, voici mon Calendly : [lien]' }
    };
  }
  return _templates;
}

module.exports = { getDmTemplates };
```

```javascript
const express = require('express');
const router = express.Router();
const { query } = require('../lib/db');
const { isAuthenticated } = require('./auth');
const { getDmTemplates } = require('../lib/dm-templates');

const STAGES = ['new', 'msg1_sent', 'msg2_sent', 'msg3_sent', 'booked', 'converted', 'lost'];

// POST /prospects/:id/advance — DM-02
router.post('/prospects/:id/advance', isAuthenticated, async (req, res, next) => {
  try {
    const { rows } = await query('SELECT id, stage FROM prospects WHERE id=$1', [req.params.id]);
    if (!rows.length) {
      req.session.flash = { type: 'error', message: 'Prospect non trouve.' };
      return res.redirect('/pipeline');
    }
    const current = rows[0].stage;
    const idx = STAGES.indexOf(current);
    if (idx === -1 || current === 'converted' || current === 'lost' || idx >= 5) {
      req.session.flash = { type: 'error', message: 'Impossible d\'avancer depuis: ' + current };
      return res.redirect('/pipeline');
    }
    const newStage = STAGES[idx + 1];
    await query('UPDATE prospects SET stage=$1 WHERE id=$2', [newStage, req.params.id]);
    req.session.flash = { type: 'success', message: 'Avance vers: ' + newStage };
    res.redirect('/pipeline');
  } catch (err) { next(err); }
});

// POST /prospects/:id/lost — DM-02 (irreversible, confirm() guarded client-side)
router.post('/prospects/:id/lost', isAuthenticated, async (req, res, next) => {
  try {
    const lost_reason = req.body.lost_reason || 'Non specifie';
    const result = await query(
      'UPDATE prospects SET stage=$1, lost_reason=$2 WHERE id=$3 RETURNING id',
      ['lost', lost_reason, req.params.id]
    );
    if (!result.rows.length) {
      req.session.flash = { type: 'error', message: 'Prospect non trouve.' };
    } else {
      req.session.flash = { type: 'success', message: 'Prospect marque comme perdu.' };
    }
    res.redirect('/pipeline');
  } catch (err) { next(err); }
});

module.exports = router;
```

```javascript
const express = require('express');
const router = express.Router();
const { query } = require('../lib/db');
const { isAuthenticated } = require('./auth');

function getISOWeek(date) {
  var d = new Date(Date.UTC(date.getFullYear(), date.getMonth(), date.getDate()));
  d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay() || 7));
  var yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
  return Math.ceil((((d - yearStart) / 86400000) + 1) / 7);
}

// GET /metrics/input — show form (STAT-01)
router.get('/metrics/input', isAuthenticated, async (req, res, next) => {
  try {
    var now = new Date();
    var week = req.query.week || (now.getFullYear() + '-W' + String(getISOWeek(now)).padStart(2, '0'));
    var { rows } = await query('SELECT * FROM metrics_weekly WHERE week=$1', [week]);
    res.render('metrics/input', { week: week, existing: rows[0] || null });
  } catch (err) { next(err); }
});

// POST /metrics/:week — upsert (STAT-01)
router.post('/metrics/:week', isAuthenticated, async (req, res, next) => {
  try {
    var week = req.params.week;
    var fields = ['posts_published','total_reach','total_impressions','total_likes',
                  'total_comments','total_shares','info_comments','dm_opened','calendly_booked'];
    var values = fields.map(function (f) { return parseInt(req.body[f] || 0, 10); });

    await query(
      'INSERT INTO metrics_weekly (week, ' + fields.join(',') + ', created_at) ' +
      'VALUES ($1,' + fields.map(function (_, i) { return '$' + (i + 2); }).join(',') + ', NOW()) ' +
      'ON CONFLICT (week) DO UPDATE SET ' +
      fields.map(function (f) { return f + '=EXCLUDED.' + f; }).join(', '),
      [week].concat(values)
    );
    req.session.flash = { type: 'success', message: 'Metriques enregistrees pour ' + week };
    res.redirect('/metrics/input?week=' + encodeURIComponent(week));
  } catch (err) { next(err); }
});

// GET /metrics/compare — STAT-04
router.get('/metrics/compare', isAuthenticated, async (req, res, next) => {
  try {
    var { rows } = await query(
      "SELECT id, hook, type, platform, published_date, " +
      "COALESCE((metrics->>'reach')::int, 0) AS reach, " +
      "COALESCE((metrics->>'likes')::int, 0) AS likes, " +
      "COALESCE((metrics->>'comments')::int, 0) AS comments_count, " +
      "COALESCE((metrics->>'shares')::int, 0) AS shares, " +
      "CASE WHEN COALESCE((metrics->>'reach')::int, 0) > 0 " +
      "  THEN ROUND(((COALESCE((metrics->>'likes')::numeric, 0) + COALESCE((metrics->>'comments')::numeric, 0)) " +
      "       / COALESCE((metrics->>'reach')::numeric, 1)) * 100, 2) " +
      "  ELSE 0 END AS engagement_rate " +
      "FROM posts WHERE status='published' ORDER BY published_date DESC NULLS LAST"
    );
    res.render('metrics/compare', { posts: rows });
  } catch (err) { next(err); }
});

module.exports = router;
```
