    ```javascript
    export function writePending({ slug, title, articlePath, imagePath }) {
      const stateDir = join(dirname(fileURLToPath(import.meta.url)), '..', 'state');
      mkdirSync(stateDir, { recursive: true });
      const pending = {
        slug,
        title,
        articlePath,
        imagePath: imagePath || null,
        timestamp: new Date().toISOString(),
        status: 'pending',
      };
      writeFileSync(join(stateDir, 'pending.json'), JSON.stringify(pending, null, 2), 'utf8');
      return pending;
    }
    ```

    **2. readPending()**
    ```javascript
    export function readPending() {
      const pendingPath = join(dirname(fileURLToPath(import.meta.url)), '..', 'state', 'pending.json');
      if (!existsSync(pendingPath)) return null;
      return JSON.parse(readFileSync(pendingPath, 'utf8'));
    }
    ```

    **3. triggerDeploy({ _deployFn, _pingFn, _basePath })**

    This is the function Phase 5 (Telegram bot) will call when Corinne taps "Approuver".

    ```javascript
    export async function triggerDeploy({ _deployFn, _pingFn, _basePath } = {}) {
      const pending = readPending();
      if (!pending) throw new Error('No pending article to deploy');

      const basePath = _basePath || process.env.SITE_BASE_PATH;
      const deploy = _deployFn || deployFiles;
      const ping = _pingFn || pingGsc;

      // Step 1: SFTP deploy (throws on failure — pending.json preserved)
      await deploy({
        slug: pending.slug,
        articlePath: pending.articlePath,
        imagePath: pending.imagePath,
        basePath,
        remoteBasePath: process.env.SFTP_REMOTE_PATH || '/',
      });
      logger.info({ slug: pending.slug }, 'SFTP deploy complete');

      // Step 2: GSC ping (graceful — does not block)
      const gscResult = await ping({
        articleUrl: `https://www.magnetiseuse-lacoste-corinne.fr/blog/${pending.slug}.html`,
      });
      logger.info({ gscStatus: gscResult.status }, 'GSC ping result');

      // Step 3: Update content-queue.yaml status: drafted -> published
      const queuePath = join(basePath, '.seo-engine', 'data', 'content-queue.yaml');
      const queueContent = readFileSync(queuePath, 'utf8');
      const idMarker = `- id: "${pending.slug}"`;
      const idx = queueContent.indexOf(idMarker);
      if (idx !== -1) {
        // Find the next status: "drafted" after this id marker and replace with "published"
        const afterId = queueContent.substring(idx);
        const updatedAfter = afterId.replace(/status:\s*"drafted"/, 'status: "published"');
        const updatedQueue = queueContent.substring(0, idx) + updatedAfter;
        writeFileSync(queuePath, updatedQueue, 'utf8');
        logger.info('content-queue.yaml status updated to published');
      }

      // Step 3b: Update content-map.yaml status: drafted -> published (per F1.8)
      const mapPath = join(basePath, '.seo-engine', 'data', 'content-map.yaml');
      const mapContent = readFileSync(mapPath, 'utf8');
      const mapIdx = mapContent.indexOf(idMarker);
      if (mapIdx !== -1) {
        const afterMapId = mapContent.substring(mapIdx);
        const publishedAt = new Date().toISOString();
        const updatedMapAfter = afterMapId
          .replace(/status:\s*"drafted"/, `status: "published"`)
          .replace(/(\s*status:\s*"published")/, `$1\n    published_at: "${publishedAt}"`);
        const updatedMap = mapContent.substring(0, mapIdx) + updatedMapAfter;
        writeFileSync(mapPath, updatedMap, 'utf8');
        logger.info('content-map.yaml status updated to published');
      }

      // Step 4: Clear pending state
      const pendingPath = join(dirname(fileURLToPath(import.meta.url)), '..', 'state', 'pending.json');
      unlinkSync(pendingPath);
      logger.info('pending.json cleared');

      return {
        slug: pending.slug,
        deployed: true,
        gscStatus: gscResult.status,
      };
    }
    ```

    Imports needed: `fs` (writeFileSync, readFileSync, mkdirSync, existsSync, unlinkSync), `path` (join, dirname), `url` (fileURLToPath), pino, and the two Plan 01 modules.

    NOTE: The `_basePath` DI parameter allows tests to use a temp directory instead of the real SITE_BASE_PATH. The `_deployFn` and `_pingFn` DI parameters replace the real sftp-deployer and gsc-ping in tests.

    For the content-queue.yaml and content-map.yaml status updates: use the SAME string-replacement pattern as file-updater.js updateSeoEngineData (NOT yaml.dump — preserves comments, per Phase 02 decision). The content-map.yaml update also inserts a `published_at` ISO timestamp on the line after `status`.
