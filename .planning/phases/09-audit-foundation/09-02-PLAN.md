    - Step 7: writes state/page-audit.json with slug-keyed entries: { score, issues, signals, previousSignals, diff, lastScanned }
    - Step 8: logs summary via pino: total pages scanned, average score, count by severity tier (green/yellow/red)
    - Returns: the full audit result object (same as what was written to file)

    Create `autopilot/audit/runner.js` as ESM module:

    ```javascript
    import { readFileSync, writeFileSync, mkdirSync } from 'node:fs';
    import { join, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';
    import pino from 'pino';
    import { buildPageInventory, normalizeSlug } from './page-inventory.js';
    import { extractPageSignals } from './signal-extractor.js';
    import { scorePageHealth } from './page-scorer.js';

    const logger = pino({ name: 'audit-runner' });
    const __dirname = dirname(fileURLToPath(import.meta.url));
    const DEFAULT_STATE_DIR = join(__dirname, '..', 'state');

    export async function runAudit({ slugs, _basePath, _readFile, _stateDir } = {}) {
      const stateDir = _stateDir || DEFAULT_STATE_DIR;
      const auditPath = join(stateDir, 'page-audit.json');

      // Step 1: Build full inventory (always full — needed for cross-page link analysis)
      logger.info('Building page inventory...');
      const inventory = buildPageInventory({ _basePath });
      const allSlugs = Object.keys(inventory);
      logger.info({ totalPages: allSlugs.length }, 'Inventory complete');

      // Step 2: Determine target slugs
      const targetSlugs = slugs
        ? slugs.filter(s => s in inventory)
        : allSlugs;
      logger.info({ targetCount: targetSlugs.length }, 'Scanning target pages');

      // Step 3: Extract signals for ALL pages (needed for internalLinkInCount)
      const allSignals = {};
      for (const slug of allSlugs) {
        try {
          allSignals[slug] = extractPageSignals(slug, inventory[slug], { _readFile });
        } catch (err) {
          logger.error({ slug, error: err.message }, 'Signal extraction failed');
          allSignals[slug] = null;
        }
      }

      // Step 4: Compute internalLinkInCount (cross-page analysis)
      // For each page, find how many other pages link to it
      const inCountMap = {};
      for (const slug of allSlugs) inCountMap[slug] = 0;

      for (const slug of allSlugs) {
        const signals = allSignals[slug];
        if (!signals) continue;
        // We need to re-parse internal link targets to map them to slugs
        // The signal extractor gives us outCount but not the actual target slugs
        // So the runner needs to do its own link-to-slug resolution
      }

      // Alternative approach: runner does its own link target extraction
      // by reading each page's HTML and finding <a href> targets, resolving to slugs
      const readFile = _readFile || readFileSync;
      const SITE_DOMAIN = 'magnetiseuse-lacoste-corinne.fr';
      const SITE_BASE_URL = `https://www.${SITE_DOMAIN}`;

      for (const slug of allSlugs) {
        if (!inventory[slug]) continue;
        try {
          const html = readFile(inventory[slug].filePath, 'utf-8');
          // Simple regex to find href values (cheerio already used in extractor, but runner can use regex for this targeted task)
          const hrefRegex = /href=["']([^"']+)["']/gi;
          let match;
          while ((match = hrefRegex.exec(html)) !== null) {
            const href = match[1];
            if (!href || href.startsWith('#') || href.startsWith('mailto:') || href.startsWith('tel:') || href.startsWith('javascript:')) continue;

            let targetSlug = null;
            try {
              const resolved = new URL(href, SITE_BASE_URL);
              if (resolved.hostname === `www.${SITE_DOMAIN}` || resolved.hostname === SITE_DOMAIN) {
                const path = resolved.pathname.replace(/^\//, '');
                targetSlug = normalizeSlug(path);
              }
            } catch {
              if (href.endsWith('.html') || href.startsWith('/') || href.startsWith('blog/')) {
                const cleaned = href.replace(/^\//, '');
                targetSlug = normalizeSlug(cleaned);
              }
            }

            if (targetSlug && targetSlug !== slug && targetSlug in inCountMap) {
              inCountMap[targetSlug]++;
            }
          }
        } catch (err) {
          logger.warn({ slug, error: err.message }, 'Failed to read HTML for link analysis');
        }
      }

      // Step 5: Load existing audit state for previousSignals/diff
      let existingState = {};
      try {
        const raw = readFile(auditPath, 'utf-8');
        existingState = JSON.parse(raw);
      } catch { /* no existing state — first run */ }

      // Step 6: Score each target page
      const results = {};
      const now = new Date().toISOString();

      for (const slug of targetSlugs) {
        const signals = allSignals[slug];
        if (!signals) {
          results[slug] = {
            score: 0,
            issues: [{ code: 'EXTRACTION_FAILED', severity: 'critical', message: 'Extraction des signaux impossible' }],
            signals: null,
            previousSignals: existingState[slug]?.signals || null,
            diff: null,
            lastScanned: now,
          };
          continue;
        }

        // Enrich signals with internalLinkInCount
        const enrichedSignals = { ...signals, internalLinkInCount: inCountMap[slug] || 0 };

        const { score, issues } = scorePageHealth(enrichedSignals);

        // Compute diff vs previous
        const previousSignals = existingState[slug]?.signals || null;
        let diff = null;
        if (previousSignals) {
          diff = {};
          for (const key of Object.keys(enrichedSignals)) {
            if (key === 'slug') continue;
            const prev = previousSignals[key];
            const curr = enrichedSignals[key];
            if (JSON.stringify(prev) !== JSON.stringify(curr)) {
              diff[key] = { before: prev, after: curr };
            }
          }
          if (Object.keys(diff).length === 0) diff = null;
        }

        results[slug] = {
          score,
          issues,
          signals: enrichedSignals,
          previousSignals,
          diff,
          lastScanned: now,
        };
      }

      // If subset scan, merge with existing results for non-target slugs
      if (slugs) {
        for (const [slug, data] of Object.entries(existingState)) {
          if (!(slug in results)) {
            results[slug] = data;
          }
        }
      }

      // Step 7: Write state file
      mkdirSync(stateDir, { recursive: true });
      writeFileSync(auditPath, JSON.stringify(results, null, 2), 'utf-8');

      // Step 8: Log summary
      const scores = Object.values(results).map(r => r.score);
      const avgScore = scores.length > 0 ? Math.round(scores.reduce((a, b) => a + b, 0) / scores.length) : 0;
      const greenCount = scores.filter(s => s >= 80).length;
      const yellowCount = scores.filter(s => s >= 50 && s < 80).length;
      const redCount = scores.filter(s => s < 50).length;

      logger.info({
        pagesScanned: targetSlugs.length,
        avgScore,
        green: greenCount,
        yellow: yellowCount,
        red: redCount,
      }, 'Audit complete');

      return results;
    }
    ```

    Key implementation details:
    - `runAudit` is async (future-proofing for Phase 10 trigger integration, even though current implementation is synchronous)
    - Cross-page internalLinkInCount: runner reads each page's HTML and extracts link targets using href regex, resolves to slugs via normalizeSlug, increments a counter map. This is separate from the signal extractor's internalLinkOutCount.
    - Subset mode (slugs param): still builds FULL inventory and extracts ALL pages for accurate internalLinkInCount, but only scores/persists the target slugs. Merges non-target slugs from existing state.
    - previousSignals and diff computed by comparing current signals with last stored signals. diff is null on first scan or when nothing changed.
    - DI: `_basePath` passed to buildPageInventory, `_readFile` passed to extractPageSignals AND used for link analysis, `_stateDir` for state file path.
    - Extraction failure for a page does not crash the entire audit — logs error and records EXTRACTION_FAILED issue.
