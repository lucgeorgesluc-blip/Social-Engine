    ```javascript
    import { readFileSync, writeFileSync, appendFileSync, mkdirSync } from 'node:fs';
    import { dirname } from 'node:path';
    import { execSync } from 'node:child_process';

    const SITE_DOMAIN = 'https://www.magnetiseuse-lacoste-corinne.fr';

    export function updateConfigJs(configPath, article) {
      const content = readFileSync(configPath, 'utf8');
      const newEntry = `        {
            slug: "${article.slug}",
            title: "${article.title}",
            description: "${article.description}",
            date: "${article.date}"
        }`;
      const updated = content.replace(
        /blog:\s*\[\s*\n/,
        `blog: [\n${newEntry},\n`
      );
      writeFileSync(configPath, updated, 'utf8');
    }

    export function rebuildConfigMin(basePath) {
      execSync(
        `npx terser assets/js/config.js -o assets/js/config.min.js -c -m`,
        { cwd: basePath, stdio: 'pipe' }
      );
    }

    export function updateSitemap(sitemapPath, slug, date) {
      const content = readFileSync(sitemapPath, 'utf8');
      const newEntry = `    <url>\n        <loc>${SITE_DOMAIN}/blog/${slug}.html</loc>\n        <lastmod>${date}</lastmod>\n        <changefreq>monthly</changefreq>\n        <priority>0.6</priority>\n    </url>`;
      const updated = content.replace(
        /(<!-- Articles blog.*?-->)\n/,
        `$1\n${newEntry}\n`
      );
      writeFileSync(sitemapPath, updated, 'utf8');
    }

    export function updateSeoEngineData(queuePath, mapPath, changelogPath, article) {
      // content-queue.yaml: targeted string replacement to preserve comments (Pitfall 6)
      let queueContent = readFileSync(queuePath, 'utf8');
      // Find the block for this article ID and change its status
      const idPattern = new RegExp(`(- id: "${article.id}"[\\s\\S]*?)status: "planned"`, 'm');
      queueContent = queueContent.replace(idPattern, '$1status: "drafted"');
      writeFileSync(queuePath, queueContent, 'utf8');

      // content-map.yaml: append new entry as YAML string (avoid full dump, preserve comments)
      const mapEntry = `\n  - slug: "${article.slug}"\n    title: "${article.title}"\n    status: "drafted"\n    target_keywords: [${article.target_keywords.map(k => `"${k}"`).join(', ')}]\n    cluster_id: "${article.cluster_id || ''}"\n    date: "${article.date}"`;
      let mapContent = readFileSync(mapPath, 'utf8');
      // Append before the last empty line or at end of blogs section
      mapContent = mapContent.trimEnd() + mapEntry + '\n';
      writeFileSync(mapPath, mapContent, 'utf8');

      // changelog.md: append timestamped entry
      const logLine = `\n- ${new Date().toISOString().slice(0, 10)} — Pipeline: drafted "${article.slug}" (${article.title})`;
      appendFileSync(changelogPath, logLine + '\n', 'utf8');
    }
    ```

    ```javascript
    import Anthropic from '@anthropic-ai/sdk';
    import pino from 'pino';

    const logger = pino({ name: 'generator' });
    const MODEL = 'claude-sonnet-4-5-20250514';

    export async function generateArticle(systemPrompt, userPrompt) {
      const client = new Anthropic(); // reads ANTHROPIC_API_KEY from env

      logger.info({ model: MODEL }, 'Starting article generation');

      const stream = client.messages.stream({
        model: MODEL,
        max_tokens: 10000,
        system: systemPrompt,
        messages: [{ role: 'user', content: userPrompt }],
      });

      let html = '';
      stream.on('text', (delta) => { html += delta; });

      const finalMsg = await stream.finalMessage();

      logger.info({
        input_tokens: finalMsg.usage.input_tokens,
        output_tokens: finalMsg.usage.output_tokens,
      }, 'Generation complete');

      return { html, usage: finalMsg.usage };
    }
    ```

    Create `autopilot/pipeline/run.js` -- the pipeline entry point invoked by `node autopilot/pipeline/run.js --dry-run`.

    ```javascript
    import { config } from 'dotenv';
    config();

    import { join } from 'node:path';
    import { writeFileSync, mkdirSync } from 'node:fs';
    import pino from 'pino';

    import { loadPipelineContext } from '../config/loader.js';
    import { COST_LOG_PATH } from '../config/constants.js';
    import { selectTopic } from './topic-selector.js';
    import { buildPrompt } from './prompt-builder.js';
    import { generateArticle } from './generator.js';
    import { validateArticle } from './validator.js';
    import { updateConfigJs, rebuildConfigMin, updateSitemap, updateSeoEngineData } from './file-updater.js';
    import { logCost, createRunGuard } from './cost-logger.js';

    const logger = pino({ name: 'pipeline' });
    const dryRun = process.argv.includes('--dry-run');

    async function main() {
      const basePath = process.env.SITE_BASE_PATH;
      if (!basePath) {
        logger.fatal('SITE_BASE_PATH env var is required');
        process.exit(1);
      }

      // Step 0: Run guard
      const guard = createRunGuard();
      if (!guard.canGenerate()) {
        logger.warn('Article limit reached for this run');
        return;
      }

      // Step 1: Load all context (F1.1)
      logger.info('Step 1: Loading pipeline context...');
      const ctx = loadPipelineContext();

      // Step 2: Select topic (F1.2)
      logger.info('Step 2: Selecting topic...');
      const { selected, reason } = selectTopic(ctx.contentQueue, ctx.contentMapTrimmed);
      if (!selected) {
        logger.warn({ reason }, 'No topic selected — stopping');
        return;
      }
      logger.info({ id: selected.id, slug: selected.slug, title: selected.title }, reason);

      // Step 3: Build prompt + generate article (F1.3)
      logger.info('Step 3: Building prompt and generating article...');
      const { system, user } = buildPrompt(ctx, selected);
      const { html, usage } = await generateArticle(system, user);

      // Step 4: Validate (F1.4)
      logger.info('Step 4: Validating generated article...');
      const validation = validateArticle(html);
      if (!validation.valid) {
        const failNames = validation.failures.map(f => f.name).join(', ');
        logger.error({ failures: failNames }, 'Validation FAILED — aborting (D-04: no retry)');
        // Log cost even on failure
        const costPath = join(basePath, 'autopilot', COST_LOG_PATH);
        logCost(usage, selected.slug, costPath);
        process.exit(1);
      }
      logger.info('Validation passed (7/7 checks)');

      // Step 5: Write article HTML to disk
      const today = new Date().toISOString().slice(0, 10);
      const articlePath = join(basePath, 'blog', `${selected.slug}.html`);
      mkdirSync(join(basePath, 'blog'), { recursive: true });
      writeFileSync(articlePath, html, 'utf8');
      logger.info({ path: articlePath }, 'Article HTML written');

      // Step 6: Update config.js + sitemap + seo-engine data (F1.5, D-07 order)
      logger.info('Step 6: Updating site files...');

      // 6a: config.js prepend
      const configPath = join(basePath, 'assets', 'js', 'config.js');
      // Extract description from the generated HTML meta description tag
      const descMatch = html.match(/<meta\s+name="description"\s+content="([^"]+)"/);
      const description = descMatch ? descMatch[1] : selected.title;
      updateConfigJs(configPath, { slug: selected.slug, title: selected.title, description, date: today });
      logger.info('config.js updated');

      // 6b: config.min.js rebuild
      if (!dryRun) {
        rebuildConfigMin(basePath);
        logger.info('config.min.js rebuilt');
      } else {
        logger.info('--dry-run: skipping config.min.js rebuild');
      }

      // 6c: sitemap.xml
      const sitemapPath = join(basePath, 'sitemap.xml');
      updateSitemap(sitemapPath, selected.slug, today);
      logger.info('sitemap.xml updated');

      // 6d: seo-engine data files
      const queuePath = join(basePath, '.seo-engine', 'data', 'content-queue.yaml');
      const mapPath = join(basePath, '.seo-engine', 'data', 'content-map.yaml');
      const changelogPath = join(basePath, '.seo-engine', 'logs', 'changelog.md');
      updateSeoEngineData(queuePath, mapPath, changelogPath, {
        id: selected.id,
        slug: selected.slug,
        title: selected.title,
        target_keywords: selected.target_keywords || [],
        cluster_id: selected.cluster_id || '',
        date: today,
      });
      logger.info('seo-engine data files updated');

      // Step 7: Log cost (F1.10)
      const costPath = join(basePath, 'autopilot', COST_LOG_PATH);
      const costEntry = logCost(usage, selected.slug, costPath);
      logger.info({ estimated_usd: costEntry.estimated_usd, input_tokens: costEntry.input_tokens, output_tokens: costEntry.output_tokens }, 'Cost logged');

      // Mark generation
      guard.recordGeneration();

      logger.info({ slug: selected.slug, dryRun }, 'Pipeline complete');
    }

    main().catch((err) => {
      logger.fatal({ err: err.message }, 'Pipeline failed');
      process.exit(1);
    });
    ```
