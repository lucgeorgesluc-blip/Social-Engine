    **Step 1: Create autopilot/audit/page-inventory.js**

    ESM module with named exports. DI via underscore-prefixed params following deploy-orchestrator.js convention.

    ```javascript
    import { readdirSync, statSync } from 'node:fs';
    import { join, relative, extname } from 'node:path';
    import { dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';
    import pino from 'pino';

    const logger = pino({ name: 'page-inventory' });
    const __dirname = dirname(fileURLToPath(import.meta.url));
    const DEFAULT_BASE_PATH = join(__dirname, '..', '..');

    const EXCLUDE_DIRS = ['node_modules', 'backup', 'logs', '.git', 'autopilot', '.planning', '.seo-engine', 'assets'];
    const EXCLUDE_FILES = ['test-config.html'];

    export function normalizeSlug(relativePath) {
      return relativePath
        .replace(/\\/g, '/')
        .replace(/^blog\//, '')
        .replace(/\.html$/, '');
    }

    export function buildPageInventory({ _basePath, _readDir, _stat } = {}) {
      const basePath = _basePath || process.env.SITE_BASE_PATH || DEFAULT_BASE_PATH;
      const readDir = _readDir || readdirSync;
      const stat = _stat || statSync;
      const inventory = {};

      // Scan root HTML files
      const rootEntries = readDir(basePath, { withFileTypes: true });
      for (const entry of rootEntries) {
        if (entry.isFile() && extname(entry.name) === '.html' && !EXCLUDE_FILES.includes(entry.name)) {
          const filePath = join(basePath, entry.name);
          const stats = stat(filePath);
          const slug = normalizeSlug(entry.name);
          inventory[slug] = { filePath, lastModified: stats.mtime.toISOString() };
        }
      }

      // Scan blog/ directory
      const blogDir = join(basePath, 'blog');
      try {
        const blogEntries = readDir(blogDir, { withFileTypes: true });
        for (const entry of blogEntries) {
          if (entry.isFile() && extname(entry.name) === '.html') {
            const filePath = join(blogDir, entry.name);
            const stats = stat(filePath);
            const slug = normalizeSlug(`blog/${entry.name}`);
            inventory[slug] = { filePath, lastModified: stats.mtime.toISOString() };
          }
        }
      } catch (err) {
        if (err.code !== 'ENOENT') throw err;
        logger.warn('blog/ directory not found — skipping');
      }

      logger.info({ pageCount: Object.keys(inventory).length }, 'Inventory built');
      return inventory;
    }

    export function pageExists(slug, inventory) {
      return slug in inventory;
    }
    ```

    Key details:
    - EXCLUDE_DIRS is not used for recursive walk — we only scan root + blog/ (per CONTEXT.md: "scans all .html files under blog/, root service pages, and index")
    - EXCLUDE_FILES contains 'test-config.html' (per robots.txt Disallow)
    - normalizeSlug handles Windows backslashes
    - pageExists takes inventory as second param (pure function, no global state)
    - DI via `{ _basePath, _readDir, _stat }` for testability

    Scoring weights (from CONTEXT.md locked decisions):
    - Schema present (LocalBusiness or HealthAndBeautyBusiness): 25pts
    - Word count >= 400: 25pts
    - FAQPage schema present: 15pts
    - E-E-A-T signal (hasAggregateRating OR hasReviewContainer): 15pts

    Issue codes and French messages:
    - `MISSING_SCHEMA` / critical / "Schema LocalBusiness ou HealthAndBeautyBusiness absent"
    - `LOW_WORD_COUNT` / critical / "Contenu trop court : {wordCount} mots (minimum 400)"
    - `MISSING_FAQ` / warning / "Schema FAQPage absent"
    - `MISSING_EEAT` / warning / "Aucun signal E-E-A-T (AggregateRating ou avis clients)"
    - `NO_INTERNAL_LINKS` / warning / "Aucun lien interne entrant depuis d'autres pages du site"
    - `MISSING_META_DESC` / critical / "Meta description absente" (when length is 0)
    - `META_DESC_LENGTH` / info / "Meta description : {length} caracteres (ideal : 120-160)" (when present but wrong length)

    ```javascript
    export function scorePageHealth(signals) {
      let score = 0;
      const issues = [];

      // Schema (25pts)
      const hasBusinessSchema = (signals.schemaTypes || []).some(t =>
        ['LocalBusiness', 'HealthAndBeautyBusiness'].includes(t)
      );
      if (hasBusinessSchema) { score += 25; }
      else { issues.push({ code: 'MISSING_SCHEMA', severity: 'critical', message: 'Schema LocalBusiness ou HealthAndBeautyBusiness absent' }); }

      // Word count (25pts)
      if ((signals.wordCount || 0) >= 400) { score += 25; }
      else { issues.push({ code: 'LOW_WORD_COUNT', severity: 'critical', message: `Contenu trop court : ${signals.wordCount || 0} mots (minimum 400)` }); }

      // FAQPage (15pts)
      if ((signals.schemaTypes || []).includes('FAQPage')) { score += 15; }
      else { issues.push({ code: 'MISSING_FAQ', severity: 'warning', message: 'Schema FAQPage absent' }); }

      // E-E-A-T (15pts)
      if (signals.hasAggregateRating || signals.hasReviewContainer) { score += 15; }
      else { issues.push({ code: 'MISSING_EEAT', severity: 'warning', message: "Aucun signal E-E-A-T (AggregateRating ou avis clients)" }); }

      // Internal links in (10pts)
      if ((signals.internalLinkInCount || 0) >= 1) { score += 10; }
      else { issues.push({ code: 'NO_INTERNAL_LINKS', severity: 'warning', message: "Aucun lien interne entrant depuis d'autres pages du site" }); }

      // Meta description (10pts)
      const mdLen = signals.metaDescriptionLength || 0;
      if (mdLen >= 120 && mdLen <= 160) { score += 10; }
      else if (mdLen === 0) { issues.push({ code: 'MISSING_META_DESC', severity: 'critical', message: 'Meta description absente' }); }
      else { issues.push({ code: 'META_DESC_LENGTH', severity: 'info', message: `Meta description : ${mdLen} caracteres (ideal : 120-160)` }); }

      return { score, issues };
    }
    ```

    Create `autopilot/audit/signal-extractor.js` as ESM module.

    ```javascript
    import { readFileSync } from 'node:fs';
    import { load } from 'cheerio';
    import pino from 'pino';

    const logger = pino({ name: 'signal-extractor' });
    const SITE_DOMAIN = 'magnetiseuse-lacoste-corinne.fr';
    const SITE_BASE_URL = `https://www.${SITE_DOMAIN}`;

    export function extractPageSignals(slug, entry, { _readFile } = {}) {
      const readFile = _readFile || readFileSync;
      const html = readFile(entry.filePath, 'utf-8');
      const $ = load(html);

      // Title
      const title = $('title').first().text().trim();

      // Meta description
      const metaDescription = $('meta[name="description"]').attr('content') || '';

      // Canonical
      const canonical = $('link[rel="canonical"]').attr('href') || null;

      // H1
      const h1Count = $('h1').length;
      const h1Text = $('h1').first().text().trim();

      // JSON-LD (FULL document scan — not head-only)
      const schemaTypes = [];
      let jsonLdBlockCount = 0;
      let jsonLdValid = true;
      let faqCount = 0;
      let hasAggregateRating = false;

      $('script[type="application/ld+json"]').each((_, el) => {
        jsonLdBlockCount++;
        try {
          const data = JSON.parse($(el).text());
          // Handle @type — can be string or array
          const types = Array.isArray(data['@type']) ? data['@type'] : (data['@type'] ? [data['@type']] : []);
          schemaTypes.push(...types);
          if (types.includes('FAQPage') && Array.isArray(data.mainEntity)) {
            faqCount = data.mainEntity.length;
          }
          if (data.aggregateRating) hasAggregateRating = true;
          // Check @graph pattern (some pages nest schemas in @graph array)
          if (Array.isArray(data['@graph'])) {
            for (const item of data['@graph']) {
              if (item['@type']) {
                const graphTypes = Array.isArray(item['@type']) ? item['@type'] : [item['@type']];
                schemaTypes.push(...graphTypes);
              }
              if (item.aggregateRating) hasAggregateRating = true;
              if (item['@type'] === 'FAQPage' && Array.isArray(item.mainEntity)) {
                faqCount = item.mainEntity.length;
              }
            }
          }
        } catch {
          jsonLdValid = false;
          logger.warn({ slug }, 'Invalid JSON-LD block');
        }
      });

      // Review container
      const hasReviewContainer = $('.review-text, .avis-client, [data-reviews]').length > 0;

      // Word count (visible body text, excluding layout elements)
      const bodyClone = $('body').clone();
      bodyClone.find('nav, header, footer, script, style, noscript').remove();
      const wordCount = bodyClone.text().split(/\s+/).filter(Boolean).length;

      // data-price count
      const dataPriceCount = $('[data-price]').length;

      // Internal links (out-count from this page)
      let internalLinkOutCount = 0;
      $('a[href]').each((_, el) => {
        const href = $(el).attr('href');
        if (!href || href.startsWith('#') || href.startsWith('mailto:') || href.startsWith('tel:')) return;
        try {
          const resolved = new URL(href, SITE_BASE_URL);
          if (resolved.hostname === `www.${SITE_DOMAIN}` || resolved.hostname === SITE_DOMAIN) {
            internalLinkOutCount++;
          }
        } catch {
          // Relative URL without protocol — if it ends in .html or starts with / or blog/, it is internal
          if (href.endsWith('.html') || href.startsWith('/') || href.startsWith('blog/')) {
            internalLinkOutCount++;
          }
        }
      });

      // Images
      const totalImages = $('img').length;
      const imagesWithAlt = $('img[alt]').filter((_, el) => $(el).attr('alt').trim() !== '').length;

      return {
        slug,
        title,
        metaDescription,
        metaDescriptionLength: metaDescription.length,
        canonical,
        h1Count,
        h1Text,
        schemaTypes,
        jsonLdBlockCount,
        jsonLdValid,
        faqCount,
        hasAggregateRating,
        hasReviewContainer,
        wordCount,
        dataPriceCount,
        internalLinkOutCount,
        totalImages,
        imagesWithAlt,
      };
    }
    ```

    Key implementation details:
    - `new URL(href, SITE_BASE_URL)` resolves relative URLs correctly (per RESEARCH.md Pitfall 5)
    - @graph pattern handled (some schemas use `{ "@graph": [...] }` wrapper)
    - @type can be string or array — both handled
    - `jsonLdValid` is false if ANY block fails to parse (not just some)
    - Word count strips nav/header/footer/script/style/noscript before counting

    **Note on internalLinkInCount:** This field requires cross-page analysis (how many OTHER pages link TO this page). It is NOT computed in extractPageSignals (which reads a single page). It will be computed in Plan 02's runner.js after all pages are extracted. The scorer's `internalLinkInCount` field will be populated by the runner before scoring. For now, the signal extractor returns `internalLinkOutCount` only.
