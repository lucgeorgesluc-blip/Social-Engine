    ```javascript
    apiRouter.get('/audit/:slug', (req, res) => {
      const auditPath = join(STATE_DIR, 'page-audit.json');
      if (!existsSync(auditPath)) return res.status(404).json({ error: 'No audit data' });
      try {
        const data = JSON.parse(readFileSync(auditPath, 'utf8'));
        const record = data[req.params.slug];
        if (!record) return res.status(404).json({ error: 'Slug not found' });

        // Enrich with cannibalization data
        let cannibalization = null;
        try {
          const allPages = Object.entries(data)
            .filter(([, v]) => v.signals)
            .map(([slug, v]) => ({ slug, title: v.signals.title || '', h1Text: v.signals.h1Text || '' }));
          const contentMap = loadContentMap();
          const pairs = detectCannibalization(allPages, contentMap.blogs || []);
          const match = pairs.find(p => p.slugA === req.params.slug || p.slugB === req.params.slug);
          if (match) {
            cannibalization = {
              slug: match.slugA === req.params.slug ? match.slugB : match.slugA,
              score: match.similarity,
              severity: match.severity,
              sharedTokens: match.sharedTokens,
            };
          }
        } catch { /* cannibalization check non-fatal */ }

        res.json({ ...record, cannibalization });
      } catch (err) {
        logger.error({ err: err.message }, 'Failed to read audit for slug');
        res.status(500).json({ error: 'Failed to read audit data' });
      }
    });
    ```

    **4. Fix test isolation in api-audit.test.js**:

    The root cause: both test files share the same AUDIT_PATH (`autopilot/state/page-audit.json`), and when run together via `node --test tests/*.test.js`, test execution order and timing cause one file's cleanup to corrupt the other's fixture.

    Fix: In `api-audit.test.js`, change the `before()` hook to save the original file at the START and the `after()` hook to ALWAYS restore it:

    ```javascript
    let originalContents = null;

    before(() => {
      mkdirSync(STATE_DIR, { recursive: true });
      if (existsSync(AUDIT_PATH)) {
        originalContents = readFileSync(AUDIT_PATH, 'utf8');
      }
    });

    after(() => {
      // Always restore original state — critical for test isolation
      if (originalContents !== null) {
        writeFileSync(AUDIT_PATH, originalContents, 'utf8');
      } else if (existsSync(AUDIT_PATH)) {
        unlinkSync(AUDIT_PATH);
      }
    });
    ```
    Add `unlinkSync` to the import from `node:fs` if not already imported.
