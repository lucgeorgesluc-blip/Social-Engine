**dashboard/lib/fb-publish.js** — Auto-publish service (CJS):

```javascript
const { query } = require('./db');
const { getPageToken } = require('./fb-token');
```

Export three functions:

1. `async function isAutoPublishEnabled()`:
   - `SELECT value FROM fb_sync_state WHERE key = 'auto_publish_enabled'`
   - Return `value === 'true'`. If no row, return `false`.

2. `async function publishSinglePost(postId)`:
   - `const token = await getPageToken()` — if null, throw `new Error('no_token')`
   - `SELECT id, content FROM posts WHERE id = $1 AND status = 'scheduled'` — if no row, throw `new Error('post_not_found')`
   - Call Graph API: `POST https://graph.facebook.com/v21.0/${token.page_id}/feed` with body `{ message: post.content, access_token: token.access_token }`
   - Use `fetch(url, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({...}) })`
   - Parse response. If `error`, throw with error message.
   - Extract `fbPostId` from response `.id`
   - Update post: `UPDATE posts SET status = 'published', published_date = NOW(), metrics = jsonb_set(COALESCE(metrics, '{}'), '{fb_post_id}', to_jsonb($1::text)) WHERE id = $2` with fbPostId and postId
   - Return `{ postId, fbPostId }`

3. `async function publishScheduledPosts()`:
   - `const enabled = await isAutoPublishEnabled()` — if false, return `{published: 0, reason: 'disabled'}`
   - `const token = await getPageToken()` — if null, return `{published: 0, error: 'no_token'}`
   - `SELECT id FROM posts WHERE status = 'scheduled' AND platform = 'facebook' AND created_date <= NOW()` (posts whose scheduled time has passed)
   - For each post, call `publishSinglePost(post.id)` wrapped in try/catch (one failure must not prevent others)
   - Return `{published: successCount, failed: failCount, errors: [...]}`

Wrap all functions in try/catch at top level — log errors, never crash.

**dashboard/routes/fb.js** — ADD settings routes (append to existing file):

```javascript
const { isAutoPublishEnabled } = require('../lib/fb-publish');
const { debugToken, getPageToken } = require('../lib/fb-token');

// GET /dashboard/settings — settings page
router.get('/dashboard/settings', isAuthenticated, async (req, res) => {
  const autoPublish = await isAutoPublishEnabled();
  const token = await getPageToken();
  let hasPublishScope = false;
  if (token) {
    const debug = await debugToken(token.access_token);
    hasPublishScope = debug && debug.scopes && debug.scopes.includes('pages_manage_posts');
  }
  res.render('settings', {
    autoPublish,
    hasPublishScope,
    hasToken: !!token
  });
});

// POST /dashboard/settings/auto-publish — toggle auto-publish
router.post('/dashboard/settings/auto-publish', isAuthenticated, async (req, res) => {
  const enable = req.body.enable === 'true';
  // Only allow enabling if token has pages_manage_posts scope
  if (enable) {
    const token = await getPageToken();
    if (!token) return res.redirect('/dashboard/settings?error=no_token');
    const debug = await debugToken(token.access_token);
    if (!debug || !debug.scopes || !debug.scopes.includes('pages_manage_posts')) {
      return res.redirect('/dashboard/settings?error=no_scope');
    }
  }
  await query(
    `INSERT INTO fb_sync_state (key, value, updated_at) VALUES ('auto_publish_enabled', $1, NOW())
     ON CONFLICT (key) DO UPDATE SET value = $1, updated_at = NOW()`,
    [enable ? 'true' : 'false']
  );
  res.redirect('/dashboard/settings');
});
```

// Auto-publish check — every 5 minutes (only if feature enabled)
cron.schedule('*/5 * * * *', async () => {
  try {
    const enabled = await isAutoPublishEnabled();
    if (!enabled) return;
    const result = await publishScheduledPosts();
    if (result.published > 0) {
      console.log('[cron] Auto-publish:', result);
    }
  } catch (err) {
    console.error('[cron] Auto-publish failed:', err.message);
  }
});
