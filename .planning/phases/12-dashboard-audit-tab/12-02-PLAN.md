    /**
     * POST /api/audit/:slug/patch — triggers patch generation for a page.
     * Calls generatePatch() from audit/patch-generator.js (Phase 11).
     * Stores result as pendingPatch in page-audit.json.
     */
    apiRouter.post('/audit/:slug/patch', async (req, res) => {
      const { slug } = req.params;
      const auditPath = join(STATE_DIR, 'page-audit.json');
      if (!existsSync(auditPath)) return res.status(404).json({ error: 'No audit data' });

      try {
        const auditData = JSON.parse(readFileSync(auditPath, 'utf8'));
        const record = auditData[slug];
        if (!record) return res.status(404).json({ error: 'Slug not found in audit data' });

        // Dynamic import — Phase 11 module may not exist yet
        let generatePatch;
        try {
          const mod = await import('../audit/patch-generator.js');
          generatePatch = mod.generatePatch;
        } catch {
          return res.status(501).json({ error: 'Patch generator not available (Phase 11 not deployed)' });
        }

        const patch = await generatePatch(slug, record.issues || []);
        // Store pendingPatch in page-audit.json
        auditData[slug].pendingPatch = patch;
        writeFileSync(auditPath, JSON.stringify(auditData, null, 2), 'utf8');
        logger.info({ slug }, 'Patch generated and stored');
        res.json({ ok: true, slug, patch });
      } catch (err) {
        logger.error({ err: err.message, slug }, 'Patch generation failed');
        res.status(500).json({ error: err.message });
      }
    });
    ```
    Note: add `writeFileSync` to the existing `import { readFileSync, existsSync, watch } from 'node:fs';` at the top of the file.

     * DELETE /api/audit/:slug/patch — rejects/removes a pending patch.
     * Clears pendingPatch field from page-audit.json for the given slug.
     */
    apiRouter.delete('/audit/:slug/patch', (req, res) => {
      const { slug } = req.params;
      const auditPath = join(STATE_DIR, 'page-audit.json');
      if (!existsSync(auditPath)) return res.status(404).json({ error: 'No audit data' });

      try {
        const auditData = JSON.parse(readFileSync(auditPath, 'utf8'));
        if (!auditData[slug]) return res.status(404).json({ error: 'Slug not found' });
        delete auditData[slug].pendingPatch;
        writeFileSync(auditPath, JSON.stringify(auditData, null, 2), 'utf8');
        logger.info({ slug }, 'Pending patch rejected');
        res.json({ ok: true, slug });
      } catch (err) {
        logger.error({ err: err.message, slug }, 'Patch rejection failed');
        res.status(500).json({ error: err.message });
      }
    });
    ```

    ```javascript
    // highlight.js safe rendering (Pitfall 1: use programmatic API, not DOM mutation)
    get patchHighlighted() {
      if (!this.selectedDetail?.pendingPatch) return '';
      if (typeof hljs === 'undefined') return this.escapeHtml(this.selectedDetail.pendingPatch);
      return hljs.highlight(this.selectedDetail.pendingPatch, { language: 'xml' }).value;
    },

    escapeHtml(str) {
      return str.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
    },

    // Signal breakdown: derive from selectedDetail.signals
    get signalList() {
      const s = this.selectedDetail?.signals;
      if (!s) return [];
      return [
        { label: 'Schema LocalBusiness', value: s.schemaTypes?.includes('HealthAndBeautyBusiness') || s.schemaTypes?.includes('LocalBusiness') ? 'Present' : 'Absent', status: s.schemaTypes?.includes('HealthAndBeautyBusiness') || s.schemaTypes?.includes('LocalBusiness') ? 'ok' : 'error' },
        { label: 'Mots (min 400)', value: (s.wordCount || 0) + ' mots', status: (s.wordCount || 0) >= 400 ? 'ok' : 'warning' },
        { label: 'Schema FAQPage', value: (s.faqCount || 0) + ' questions', status: (s.faqCount || 0) > 0 ? 'ok' : 'warning' },
        { label: 'E-E-A-T (avis/note)', value: s.hasReviewSignal ? 'Present' : 'Absent', status: s.hasReviewSignal ? 'ok' : 'warning' },
        { label: 'Liens internes entrants', value: String(s.internalLinkInCount ?? 0), status: (s.internalLinkInCount ?? 0) >= 1 ? 'ok' : 'error' },
        { label: 'Meta description', value: s.metaDescriptionLength ? s.metaDescriptionLength + ' car.' : 'Absente', status: s.metaDescriptionLength >= 120 && s.metaDescriptionLength <= 160 ? 'ok' : (s.metaDescriptionLength ? 'warning' : 'error') },
        { label: 'Canonical', value: s.hasCanonical ? 'Present' : 'Absent', status: s.hasCanonical ? 'ok' : 'error' },
        { label: 'H1', value: s.h1Text || 'Absent', status: s.h1Text ? 'ok' : 'error' },
      ];
    },

    // Issues sorted by severity: critical first, then warning, then info
    get sortedIssues() {
      if (!this.selectedDetail?.issues) return [];
      const issues = [...this.selectedDetail.issues];
      const order = { critical: 0, warning: 1, info: 2 };
      return issues
        .filter(i => typeof i === 'object')
        .sort((a, b) => (order[a.severity] ?? 3) - (order[b.severity] ?? 3));
    },

    // Confirm modal for patch apply (reuses existing _confirmModal store)
    confirmApplyPatch(slug) {
      Alpine.store('nav')._confirmModal = {
        open: true,
        slug,
        title: 'Appliquer le patch SEO sur ' + slug + ' ?'
      };
      // Listen for confirm event
      const handler = async () => {
        window.removeEventListener('confirm-action', handler);
        window.removeEventListener('cancel-action', cancelHandler);
        await this.applyPatch(slug);
      };
      const cancelHandler = () => {
        window.removeEventListener('confirm-action', handler);
        window.removeEventListener('cancel-action', cancelHandler);
      };
      window.addEventListener('confirm-action', handler, { once: true });
      window.addEventListener('cancel-action', cancelHandler, { once: true });
    },
    ```
