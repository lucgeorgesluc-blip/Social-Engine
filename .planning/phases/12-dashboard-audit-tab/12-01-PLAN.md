    **6. Register `auditPanel` Alpine component** inside the `alpine:init` listener, BEFORE the closing `});` of the listener (after the `costsPanel` registration around line 1710):
    ```javascript
    Alpine.data('auditPanel', () => ({
      loading: true,
      error: '',
      pages: [],
      chutes: [],
      selectedAuditSlug: null,
      selectedDetail: null,
      detailLoading: false,
      generating: new Set(),
      applying: new Set(),
      auditing: false,
      sortBy: 'score-asc',

      async init() {
        await this.loadAudit();
        this.loading = false;
        // Listen for SSE refresh
        window.addEventListener('audit-complete', () => this.loadAudit());
      },

      async loadAudit() {
        try {
          const data = await apiFetch('/api/audit');
          if (!data) return;
          // Convert slug-keyed object to array (Pitfall 6 from RESEARCH.md)
          if (data && typeof data === 'object' && !Array.isArray(data)) {
            const pagesObj = data.pages && typeof data.pages === 'object' ? data : { pages: data };
            const source = pagesObj.pages && Object.keys(pagesObj.pages).length === 0 ? data : (pagesObj.pages || data);
            this.pages = Object.entries(source)
              .filter(([k]) => k !== 'pages')
              .map(([slug, d]) => ({
                slug,
                score: d.score ?? 0,
                issues: d.issues || [],
                criticalCount: (d.issues || []).filter(i => (typeof i === 'object' ? i.severity : '') === 'critical').length,
                warningCount: (d.issues || []).filter(i => (typeof i === 'object' ? i.severity : '') === 'warning').length,
                lastScanned: d.lastScanned || '',
                pendingPatch: d.pendingPatch || null,
              }));
          }
          // Load chutes from audit-status.json (via a separate field or from the response)
          // audit-status.json is read separately since GET /api/audit returns page-audit.json only
          try {
            const statusData = await apiFetch('/api/audit-status');
            if (statusData && statusData.triggeredAt) {
              this.chutes = [statusData];
            } else {
              this.chutes = [];
            }
          } catch {
            this.chutes = [];
          }
          // Update critical count badge in sidebar
          const critTotal = this.pages.reduce((sum, p) => sum + p.criticalCount, 0);
          Alpine.store('nav').auditCriticalCount = critTotal;
          // Refresh stats (avgSeoScore)
          try {
            const statsData = await apiFetch('/api/stats');
            if (statsData) {
              const app = document.querySelector('[x-data="dashboardApp()"]');
              if (app && app.__x) {
                app.__x.$data.stats = { ...app.__x.$data.stats, ...statsData };
              }
            }
          } catch { /* non-critical */ }
        } catch (err) {
          this.error = "Erreur de chargement de l'audit.";
        }
      },

      async selectPage(slug) {
        if (this.selectedAuditSlug === slug) {
          this.selectedAuditSlug = null;
          this.selectedDetail = null;
          return;
        }
        this.selectedAuditSlug = slug;
        this.detailLoading = true;
        this.selectedDetail = null;
        try {
          const data = await apiFetch('/api/audit/' + slug);
          this.selectedDetail = data;
        } catch {
          this.selectedDetail = null;
        } finally {
          this.detailLoading = false;
        }
      },

      async generatePatch(slug) {
        this.generating = new Set([...this.generating, slug]);
        try {
          await apiFetch('/api/audit/' + slug + '/patch', { method: 'POST' });
          await this.loadAudit();
          if (this.selectedAuditSlug === slug) {
            await this.selectPage(slug); // close
            await this.selectPage(slug); // reopen with fresh data
          }
        } catch (err) {
          this.error = 'La generation du patch a echoue. Verifiez que le service Claude API est disponible.';
        } finally {
          const next = new Set(this.generating);
          next.delete(slug);
          this.generating = next;
        }
      },

      async applyPatch(slug) {
        // Will be fully implemented in Plan 02 with confirm modal
        this.applying = new Set([...this.applying, slug]);
        try {
          await apiFetch('/api/audit/' + slug + '/apply', { method: 'POST' });
          await this.loadAudit();
        } catch (err) {
          this.error = 'Echec du deploiement. Le fichier original a ete restaure automatiquement.';
        } finally {
          const next = new Set(this.applying);
          next.delete(slug);
          this.applying = next;
        }
      },

      async rejectPatch(slug) {
        try {
          await apiFetch('/api/audit/' + slug + '/patch', { method: 'DELETE' });
          await this.loadAudit();
          if (this.selectedAuditSlug === slug) {
            await this.selectPage(slug);
            await this.selectPage(slug);
          }
        } catch {
          this.error = 'Erreur lors du rejet du patch.';
        }
      },

      async runAudit() {
        this.auditing = true;
        this.error = '';

        } catch {
          this.error = "L'audit a echoue. Verifiez les logs du serveur et reessayez.";
          this.auditing = false;
        }
      },

      scoreBadgeClass(score) {
        if (score >= 80) return 'bg-[#22c55e] text-[#0d1117]';
        if (score >= 60) return 'bg-[#eab308] text-[#0d1117]';

      severityBadgeClass(severity) {

          warning: 'bg-[#eab308]/20 text-[#eab308]',
          info: 'bg-[#3b82f6]/20 text-[#3b82f6]',
          healthy: 'bg-[#22c55e]/20 text-[#22c55e]',
        };
        return 'inline-flex items-center px-2 py-0.5 rounded-full text-xs font-semibold ' + (map[severity] || map.info);
      },

      get sortedPages() {

        if (this.sortBy === 'score-asc') arr.sort((a, b) => a.score - b.score);
        else if (this.sortBy === 'score-desc') arr.sort((a, b) => b.score - a.score);
        else arr.sort((a, b) => a.slug.localeCompare(b.slug));
    ```
