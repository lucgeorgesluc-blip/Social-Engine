    ```javascript
    import { appendFileSync, readFileSync, writeFileSync, existsSync, mkdirSync } from 'node:fs';
    import { join, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';
    import pino from 'pino';

    const logger = pino({ name: 'activity-logger' });
    const __dirname = dirname(fileURLToPath(import.meta.url));
    const STATE_DIR = join(__dirname, '..', 'state');
    const activityPath = join(STATE_DIR, 'activity.jsonl');
    const statusPath = join(STATE_DIR, 'pipeline-status.json');

    const STEPS = [
      { step: 1, name: 'Read Context' },
      { step: 2, name: 'Pick Topic' },
      { step: 3, name: 'Draft' },
      { step: 4, name: 'Generate Image' },
      { step: 5, name: 'Await Approval' },
      { step: 6, name: 'Deploy' },
    ];

    export function writeActivityEvent(event, { _appendFn, _activityPath } = {}) {
      const append = _appendFn || appendFileSync;
      const path = _activityPath || activityPath;
      mkdirSync(dirname(path), { recursive: true });
      const line = JSON.stringify({ ...event, ts: event.ts || new Date().toISOString() }) + '\n';
      append(path, line, 'utf8');
      logger.info({ event: event.event, slug: event.slug }, 'Activity event written');
    }

    export function readRecentActivity(count = 20, { _readFn, _activityPath } = {}) {
      const read = _readFn || readFileSync;
      const path = _activityPath || activityPath;
      try {
        const content = read(path, 'utf8');
        return content.trim().split('\n').filter(Boolean).slice(-count).reverse().map(line => {
          try { return JSON.parse(line); } catch { return null; }
        }).filter(Boolean);
      } catch { return []; }
    }

    export function formatActivity(event) {
      switch (event.event) {
        case 'drafted':
          return `Article rédigé : ${event.slug} (${event.wordCount} mots, ${event.linkCount} liens)`;
        case 'telegram_sent':
          return `Envoyé sur Telegram : ${event.slug}`;
        case 'deployed':
          return `Déployé : ${event.slug} → ${event.url}`;
        case 'error':
          return `Erreur (${event.step}) : ${event.message}`;
        default:
          return `${event.event}: ${event.slug || ''}`;
      }
    }

    export function writePipelineStatus(stepNumber, { _writeFn, _readFn, _statusPath } = {}) {
      const write = _writeFn || writeFileSync;
      const read = _readFn || readFileSync;
      const path = _statusPath || statusPath;
      mkdirSync(dirname(path), { recursive: true });
      let existing = { history: [] };
      if (existsSync(path)) {
        try { existing = JSON.parse(read(path, 'utf8')); } catch { existing = { history: [] }; }
      }
      const stepDef = STEPS.find(s => s.step === stepNumber) || { step: stepNumber, name: `Step ${stepNumber}` };
      const now = new Date().toISOString();
      const status = {
        step: stepDef.step,
        stepName: stepDef.name,
        ts: now,
        history: [...(existing.history || []), { step: stepDef.step, stepName: stepDef.name, ts: now }],
      };
      write(path, JSON.stringify(status, null, 2), 'utf8');
    }
    ```
