    **Step 1: Create `autopilot/telegram/edit-handler.js`** (new file):

    ```javascript
    import Anthropic from '@anthropic-ai/sdk';
    import pino from 'pino';

    const logger = pino({ name: 'edit-handler' });
    const EDIT_MODEL = 'claude-sonnet-4-5';

    const EDIT_SYSTEM_PROMPT = `You are editing an existing French HTML article for a magnetiseuse website.
    Return ONLY the complete modified HTML file. No explanations, no markdown fences.

    RULES you MUST follow:
    - Never use hard-coded prices (no euro amounts). Use data-price="tabac" or other data-price attributes.
    - Never mention rTMS or stimulation magnetique transcranienne.
    - Keep all existing FAQPage schema markup intact.
    - Keep all data-blog-list="related" blocks intact.
    - Keep the <link rel="canonical"> tag intact.
    - Keep data-price= attributes on all price elements.`;

    export async function editArticle(existingHtml, editInstruction, { _client } = {}) {
      const client = _client || new Anthropic();

      logger.info({ instructionLength: editInstruction.length }, 'Starting edit-in-place via Claude');

      const stream = client.messages.stream({
        model: EDIT_MODEL,
        max_tokens: 16000,
        system: EDIT_SYSTEM_PROMPT,
        messages: [{
          role: 'user',
          content: `<existing_html>${existingHtml}</existing_html>\n<edit_instruction>${editInstruction}</edit_instruction>\nReturn only the modified complete HTML file.`,
        }],
      });

      let html = '';
      stream.on('text', (delta) => { html += delta; });
      const finalMsg = await stream.finalMessage();

      logger.info({
        input_tokens: finalMsg.usage.input_tokens,
        output_tokens: finalMsg.usage.output_tokens,
      }, 'Edit-in-place complete');

      return { html, usage: finalMsg.usage };
    }
    ```
