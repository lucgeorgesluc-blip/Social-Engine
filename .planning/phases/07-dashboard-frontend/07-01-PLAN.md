// Helper: returns status badge HTML string (used via x-html or factored as class helper)
function statusBadgeClass(status) {
  const map = {
    published:  'bg-[#22c55e] text-[#0d1117]',
    pending:    'bg-[#eab308] text-[#0d1117]',
    drafted:    'bg-[#3b82f6] text-white',
    queued:     'bg-[#6b7280] text-white',
  };
  return 'inline-flex items-center px-2 py-0.5 rounded-full text-xs font-semibold ' + (map[status] || map.queued);
}

function statusLabel(status) {
  const labels = { published: 'Publié', pending: 'En attente', drafted: 'Rédigé', queued: 'En file' };
  return labels[status] || status;
}

async function apiFetch(path) {
  const r = await fetch(path, { credentials: 'same-origin' });
  if (r.status === 401) { window.location.href = '/login.html'; return null; }
  if (!r.ok) throw new Error(`HTTP ${r.status}`);
  return r.json();
}

async function logout() {
  await fetch('/logout', { method: 'POST', credentials: 'same-origin' });
  window.location.href = '/login.html';
}

// SSE connection (called once after Alpine init)
let evtSource = null;
function connectSSE() {
  evtSource = new EventSource('/api/events');
  evtSource.onopen = () => { Alpine.store('nav').sseStatus = 'connected'; };
  evtSource.onmessage = (e) => {
    try {
      const msg = JSON.parse(e.data);
      if (msg.type === 'pipeline') {
        // Pipeline panel will react via its own x-data listening to store updates
        Alpine.store('nav')._pipelinePayload = msg.payload;
      }
      if (msg.type === 'pending') {
        Alpine.store('nav').pendingCount = Array.isArray(msg.payload) ? msg.payload.length : (msg.payload || 0);
      }
    } catch {}
  };
  evtSource.onerror = () => {
    Alpine.store('nav').sseStatus = 'disconnected';
    evtSource.close();
    evtSource = null;
    // Before reconnecting, health check to detect 401 vs network issue
    fetch('/health').then(r => {
      if (r.status === 401) { window.location.href = '/login.html'; return; }
      setTimeout(connectSSE, 3000);
    }).catch(() => setTimeout(connectSSE, 3000));
  };
}

document.addEventListener('alpine:init', () => {
  Alpine.store('nav', { tab: 'queue', pendingCount: 0, sseStatus: 'disconnected', _pipelinePayload: null });

  Alpine.data('dashboardApp', () => ({
    stats: {},
    activity: [],
    loading: true,
    error: '',
    articles: [],
    search: '',
    approving: new Set(),
    steps: [
      { name: 'Lecture du contexte',      status: 'pending', timestamp: '' },
      { name: 'Sélection du sujet',       status: 'pending', timestamp: '' },
      { name: 'Rédaction de l\'article',  status: 'pending', timestamp: '' },
      { name: 'Génération de l\'image',   status: 'pending', timestamp: '' },
      { name: 'Validation + Approbation', status: 'pending', timestamp: '' },
      { name: 'Déploiement SFTP',         status: 'pending', timestamp: '' },
    ],

    async init() {
      connectSSE();
      await Promise.all([this.loadStats(), this.loadActivity(), this.loadArticles()]);
      this.loading = false;
    },

    async loadStats() {
      try {
        const data = await apiFetch('/api/stats');
        if (data) {
          this.stats = data;
          Alpine.store('nav').pendingCount = data.pending || 0;
        }
      } catch { this.error = 'Erreur de chargement des statistiques.'; }
    },

    async loadActivity() {
      try {
        const data = await apiFetch('/api/activity');
        if (data) this.activity = data.events || [];
      } catch {}
    },

    async loadArticles() {
      try {
        const data = await apiFetch('/api/articles');
        if (data) this.articles = data.articles || [];
      } catch { this.error = 'Erreur de chargement des articles.'; }
    },

    get filteredArticles() {
      if (!this.search) return this.articles;
      const q = this.search.toLowerCase();
      return this.articles.filter(a => a.title.toLowerCase().includes(q) || a.slug.toLowerCase().includes(q));
    },

    statusBadgeClass,
    statusLabel,

    async approve(slug) {
      this.approving = new Set([...this.approving, slug]);
      try {
        const r = await fetch(`/api/articles/${slug}/approve`, {
          method: 'POST',
          credentials: 'same-origin',
          headers: { 'Content-Type': 'application/json' }
        });
        if (r.ok) {
          const a = this.articles.find(x => x.slug === slug);
          if (a) a.status = 'published';
          Alpine.store('nav').pendingCount = Math.max(0, (Alpine.store('nav').pendingCount || 0) - 1);
        } else {
          alert('Erreur lors du déploiement. Réessayez ou vérifiez les logs.');
        }
      } catch {
