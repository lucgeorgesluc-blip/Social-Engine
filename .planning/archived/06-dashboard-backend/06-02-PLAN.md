    ```javascript
    import { google } from 'googleapis';
    import { readFileSync, writeFileSync, existsSync, mkdirSync } from 'node:fs';
    import { join, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';
    import pino from 'pino';

    const logger = pino({ name: 'gsc-rankings' });
    const __dirname = dirname(fileURLToPath(import.meta.url));
    const SITE_URL = 'https://www.magnetiseuse-lacoste-corinne.fr/';
    const CACHE_TTL_MS = 60 * 60 * 1000; // 1 hour
    const DEFAULT_CACHE_PATH = join(__dirname, '..', 'state', 'rankings-cache.json');

    function loadHighPriorityKeywords() {
      const basePath = process.env.SITE_BASE_PATH || '.';
      const csvPath = join(basePath, '.seo-engine', 'data', 'seo-keywords.csv');
      if (!existsSync(csvPath)) return [];
      const lines = readFileSync(csvPath, 'utf8').trim().split('\n');
      if (lines.length < 2) return [];
      const headers = lines[0].split(',').map(h => h.trim().replace(/^"|"$/g, ''));
      const kwIdx = headers.indexOf('keyword');
      const prIdx = headers.indexOf('priority');
      if (kwIdx === -1 || prIdx === -1) return [];
      return lines.slice(1)
        .map(line => {
          const cols = line.split(',').map(c => c.trim().replace(/^"|"$/g, ''));
          return { keyword: cols[kwIdx], priority: cols[prIdx] };
        })
        .filter(r => r.priority === 'high')
        .map(r => r.keyword)
        .filter(Boolean);
    }

    export async function getRankings({ period = '30d', cachePath, _searchConsole } = {}) {
      const cPath = cachePath || DEFAULT_CACHE_PATH;

      // Serve from cache if fresh
      if (existsSync(cPath)) {
        try {
          const cache = JSON.parse(readFileSync(cPath, 'utf8'));
          if (Date.now() - cache.fetchedAt < CACHE_TTL_MS && cache.period === period) {
            logger.info('Serving rankings from cache');
            return cache.data;
          }
        } catch { /* stale/corrupt cache — refetch */ }
      }

      if (!process.env.GSC_SERVICE_ACCOUNT_PATH) {
        logger.warn('GSC_SERVICE_ACCOUNT_PATH not set -- skipping rankings fetch');
        return { error: 'GSC not configured', rows: [] };
      }

      const days = period === '7d' ? 7 : period === '90d' ? 90 : 30;
      const endDate = new Date().toISOString().slice(0, 10);
      const startDate = new Date(Date.now() - days * 86400000).toISOString().slice(0, 10);

      try {
        const auth = new google.auth.GoogleAuth({
          keyFile: process.env.GSC_SERVICE_ACCOUNT_PATH,
          scopes: ['https://www.googleapis.com/auth/webmasters.readonly'],
        });
        const sc = _searchConsole || google.searchconsole({ version: 'v1', auth });

        const response = await sc.searchanalytics.query({
          siteUrl: SITE_URL,
          requestBody: {
            startDate,
            endDate,
            dimensions: ['date', 'query'],
            rowLimit: 25000,
            dataState: 'all',
          },
        });

        const rows = response.data.rows || [];
        const result = { rows, keywords: loadHighPriorityKeywords(), period, startDate, endDate };

        // Write cache
        mkdirSync(dirname(cPath), { recursive: true });
        writeFileSync(cPath, JSON.stringify({ fetchedAt: Date.now(), period, data: result }), 'utf8');

        logger.info({ rowCount: rows.length, period }, 'GSC rankings fetched and cached');
        return result;
      } catch (err) {
        logger.warn({ err: err.message }, 'GSC rankings fetch failed -- graceful fallback');
        return { error: err.message, rows: [], period };
      }
    }
    ```

    ```javascript
    import { Router } from 'express';
    import { readFileSync, existsSync, watch } from 'node:fs';
    import { join, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';
    import yaml from 'js-yaml';
    import pino from 'pino';
    import { triggerDeploy, readPendingArray } from '../pipeline/deploy-orchestrator.js';
    import { getRankings } from '../pipeline/gsc-rankings.js';
    import { readRecentActivity, formatActivity } from '../pipeline/activity-logger.js';

    const logger = pino({ name: 'api-routes' });
    const __dirname = dirname(fileURLToPath(import.meta.url));
    const STATE_DIR = join(__dirname, '..', 'state');
    const PIPELINE_STATUS_PATH = join(STATE_DIR, 'pipeline-status.json');
    const PENDING_PATH = join(STATE_DIR, 'pending.json');

    export const apiRouter = Router();

    // --- Helper: load content-map.yaml ---
    function loadContentMap() {
      const basePath = process.env.SITE_BASE_PATH || '.';
      const mapPath = join(basePath, '.seo-engine', 'data', 'content-map.yaml');
      if (!existsSync(mapPath)) return { blogs: [] };
      try {
        return yaml.load(readFileSync(mapPath, 'utf8')) || { blogs: [] };
      } catch { return { blogs: [] }; }
    }

    // --- Helper: build D3 link tree ---
    export function buildLinkTree(contentMap) {
      const tree = { name: 'magnetiseuse-lacoste-corinne.fr', type: 'root', children: [] };
      const blogs = contentMap.blogs || [];
      const clusters = {};
      for (const blog of blogs) {
        const cid = blog.cluster_id || 'unclustered';
        if (!clusters[cid]) clusters[cid] = [];
        clusters[cid].push(blog);
      }
      for (const [clusterId, pages] of Object.entries(clusters)) {
        const pillar = pages.find(p => p.content_type === 'pillar');
        tree.children.push({
          name: pillar?.title || clusterId,
          type: 'pillar',
          slug: pillar?.slug || clusterId,
          children: pages.filter(p => p !== pillar).map(p => ({
            name: p.title,
            type: p.content_type || 'blog',
            slug: p.slug,
            links_to: p.internal_links_to || [],
            children: [],
          })),
        });
      }
      const blogSlugs = new Set(blogs.map(b => b.slug));
      const externalRefs = new Set();
      for (const blog of blogs) {
        for (const link of (blog.internal_links_to || [])) {
          const slug = link.replace(/\.html$/, '');
          if (!blogSlugs.has(slug)) externalRefs.add(slug);
        }
      }
      if (externalRefs.size > 0) {
        tree.children.push({
          name: 'Pages Services', type: 'service',
          children: [...externalRefs].map(slug => ({ name: slug, type: 'service', slug, children: [] })),
        });
      }
      return tree;
    }

    // --- Helper: map article status ---
    function mapStatus(blog, pendingSlugs) {
      if (pendingSlugs.has(blog.slug)) return 'pending';
      if (blog.status === 'published') return 'published';
      if (blog.status === 'drafted') return 'drafted';
      return 'queued';
    }

    // ─── Routes ──────────────────────────────────────────────────────────────

    // GET /api/articles
    apiRouter.get('/articles', (req, res) => {
      const contentMap = loadContentMap();
      const pending = readPendingArray();
      const pendingSlugs = new Set(pending.map(p => p.slug));
      let articles = (contentMap.blogs || []).map(blog => ({
        slug: blog.slug,
        title: blog.title || blog.slug,
        status: mapStatus(blog, pendingSlugs),
        wordCount: blog.word_count || 0,
        linkCount: (blog.internal_links_to || []).length,
        imageStatus: blog.image_status || 'unknown',
        date: blog.published_at || '',
      }));
      if (req.query.status === 'pending') {
        articles = articles.filter(a => a.status === 'pending');
      }
      res.json({ articles });
    });

    // GET /api/rankings
    apiRouter.get('/rankings', async (req, res) => {
      try {
        const data = await getRankings({ period: req.query.period || '30d' });
        res.json(data);
      } catch (err) {
        logger.error({ err: err.message }, 'Rankings fetch error');
        res.status(500).json({ error: err.message, rows: [] });
      }
    });

    // GET /api/links
    apiRouter.get('/links', (req, res) => {
      const contentMap = loadContentMap();
      res.json(buildLinkTree(contentMap));
    });

    // GET /api/pipeline-status
    apiRouter.get('/pipeline-status', (req, res) => {
      if (!existsSync(PIPELINE_STATUS_PATH)) {
        return res.json({ step: 0, stepName: 'Idle', ts: null, history: [] });
      }
      try {
        res.json(JSON.parse(readFileSync(PIPELINE_STATUS_PATH, 'utf8')));
      } catch {
        res.json({ step: 0, stepName: 'Idle', ts: null, history: [] });
      }
    });

    // GET /api/events (SSE)
    apiRouter.get('/events', (req, res) => {
      res.writeHead(200, {
        'Content-Type': 'text/event-stream',
        'Cache-Control': 'no-cache',
        Connection: 'keep-alive',
      });
      res.flushHeaders();
      res.write(`data: ${JSON.stringify({ type: 'connected' })}\n\n`);

      // Debounce factory (Pitfall 6: Windows double-fire)
      function debounced(fn, ms = 100) {
        let timer = null;
        return () => { clearTimeout(timer); timer = setTimeout(fn, ms); };
      }

      const watchers = [];
      function safeWatch(filePath, type) {
        if (!existsSync(filePath)) return;
        try {
          const watcher = watch(filePath, debounced(() => {
            try {
              const data = JSON.parse(readFileSync(filePath, 'utf8'));
              res.write(`data: ${JSON.stringify({ type, payload: data })}\n\n`);
            } catch { /* file mid-write or removed */ }
          }));
          watchers.push(watcher);
        } catch { /* file watch failed — non-fatal */ }
      }

      safeWatch(PIPELINE_STATUS_PATH, 'pipeline');
      safeWatch(PENDING_PATH, 'pending');

      const keepalive = setInterval(() => res.write(': keepalive\n\n'), 30000);

      req.on('close', () => {
        clearInterval(keepalive);
        watchers.forEach(w => { try { w.close(); } catch {} });
      });
    ```
